汉诺塔

汉诺塔相信大家都略有耳闻，这里就不多解释；

数据结构课上，讲到了汉诺塔问题——“利用递归求解移动过程”，那我就不得不拿出来以前的代码（不，还是自己

回忆写一下比较好😄），这个其实早就学过了，之前刷acm的题目，做过好几道汉诺塔问题，但是没这么简单😭

话不多说，直接上code：

#include <stdio.h>

int count = 0; //全局变量，计数
int move(char x,char y)
{
     printf("%c->%c\n",x,y); //x柱拿一个圆盘移动到y柱
}
int hanoi(int n,char a,char b,char c)
{
    if(n==1) //只有一个圆盘
    {
        move(a,c); //直接从a移动到c
        count ++;
    }
    else
    {
        hanoi(n-1,a,c,b); //n-1个圆盘从a借助c移动到b
        move(a,c); //a中剩下的一个圆盘移动到c
        count ++;
        hanoi(n-1,b,a,c); //移动到b的n-1个圆盘从b借助a移动到c
    }
}

int main()
{
    int n;
    printf("请输入圆盘数：");
    scanf("%d",&n);
    hanoi(n,'A','B','C');
    printf("\n需要移动的次数为：%d",count);
    return 0;
}

汉诺塔问题算是纯递归的啦，很简单呐😜

杨辉三角

队列实现

这个是老师留的作业，哇，好难写，这真够我写一晚上的啦

#include<stdio.h>
#include<stdlib.h>
#define MAXSIZE 1000 

typedef int ElemType;
typedef struct
{
    ElemType data[MAXSIZE];
    int front,rear;
}Queue;

void InitQueue(Queue *Q) //初始化队列
{
    Q->front = Q->rear = -1;
}

int EmptyQueue(Queue *Q) //判断队列是否为空
{
    return Q->rear == Q->front;
}

void EnQueue(Queue *Q, ElemType e) //入队 
{
    if(Q->rear == MAXSIZE)
    {
        return;
    }
    Q->data[Q->rear] = e;
    Q->rear ++;
}

void DeQueue(Queue *Q,ElemType *e) //出队
{
    if(Q->rear == Q->front)
    {
        return;
    }
    *e = Q->data[Q->front];
    Q->front ++;
}

void GetHeadData(Queue *Q,ElemType *e) //获取队头元素
{
    if(EmptyQueue(Q))
    {
        return;
    }
    else
    {
        *e = Q->data[Q->front];
    }
}

void yanghui(int n) //杨辉三角实现函数
{
    Queue Q;
    int i,x,e,k=0;
    for(i=1;i<=n;i++)
    {
        printf("   ");  //单独打印第一行数字1 ，需要控制格式 
    }   
    printf("1\n"); // print(1) 

    InitQueue(&Q);
    EnQueue(&Q,0);
    EnQueue(&Q,1);
    EnQueue(&Q,1); //队列初始化 
    while(++k<n) //输出后n-1行 
    {
        for(i=1;i<=n-k;i++)
        {
            printf("   "); //控制格式，采取"3的倍数格式控制" 
        }
        EnQueue(&Q,0); //入队0，为while和if判断 
        e = 1;
        while(e!=0)
        {
            DeQueue(&Q,&x);
            GetHeadData(&Q,&e);
            if(e)
            {
                printf("%-6d",e);
            }
            else
            {
                printf("\n");
            }
            EnQueue(&Q,x+e); //入队->每个数等于它上方两数之和 
        }
    }
}
int main()
{
    int n;
    printf("请输入需要打印的行数：");
    scanf("%d",&n); 
    printf("\n");
    yanghui(n);
    printf("\n");

    return 0;
}

C语言实现

杨辉三角用C语言打印更简单！！🙃

#include <stdio.h>

int main()
{
    int n,i,j,k,a[100][100];  
    printf("请输入需要打印的行数：");
    scanf("%d",&n);
    printf("\n");

    for(i=1;i<=n;i++)
    {
        a[i][1] = a[i][i] = 1;  //每一行第一个和最后一个数字为1 
    } 

    for(i=3;i<=n;i++) //这里从第三行开始，因为前两行全部是数字1，已经赋值完毕 
    {
        for(j=2;j<=i-1;j++) //每一行从第二个数字开始 
        {
            a[i][j]=a[i-1][j-1]+a[i-1][j]; //每个数字等于它上方两数之和 
        }
    }            
    for(i=1;i<=n;i++) //从第一行开始输出 
    {
        for(k=1;k<=n-i+1;k++) //控制格式 
        {
            printf("   "); 
        }

        for(j=1;j<=i;j++)  
        {
            printf("%-6d",a[i][j]); //左对齐，输出 
        } 
        printf("\n");  
    }
    printf("\n");

    return 0; 
}
