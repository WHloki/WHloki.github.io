sort

先来一个简单的sort吧；

sort() 是C++标准库里面的库函数，之前做编程练习题目的时候经常会用到，算是quicksort的另一种形式，因为时

间复杂度都为 O(log₂N)，完美地解决了所谓的TLE问题，直接上代码：

#include<iostream>
#include<algorithm>
using namespace std;

int main()
{
    int a[100],i,n;
    cin >> n;    
    for(i=0;i<n;i++)
    {
        cin >> a[i];
    }
    sort(a,a+n); //调用sort函数 
    for(i=0;i<n;i++)
    {
        cout << a[i] << ' ';
    }
}

这种是最简单的形式，默认从小到大，那从大到小呢？嘿嘿，别着急，来看：

#include<iostream>
#include<algorithm>
using namespace std;

int cmp(int m,int n)  //自定义函数 
{
    return m>n;
} 
int main()
{
    int a[100],i,n;
    cin >> n;    
    for(i=0;i<n;i++)
    {
        cin >> a[i];
    }
    sort(a,a+n,cmp); //调用sort函数 
    for(i=0;i<n;i++)
    {
        cout << a[i] << ' ';
    }
}

这种是加一个自定义函数cmp，用于大小比较，从小到大时也可以自定义一个函数，但是没必要其实；

先看最原始的代码（我第一次写的代码）：

#include<iostream>
#include<algorithm>
using namespace std;

int gcd(int a,int b)
{
    int t;
    while(b!=0)
    {
        t = a%b; 
        a = b;
         b = t;
    }
    return a;
}

int main()
{
    int m,n;
    cin >> m >> n;
    cout << gcd(m,n) << endl;
}

lcm

lcm其实没啥好说的，用三目来说，直接看代码：

#include<iostream>
#include<algorithm>
using namespace std;

int gcd(int a,int b)
{
    return !b?a:gcd(b,a%b);
}
int lcm(int m,int n)
{
    return (m*n)/gcd(m,n);
} 
int main()
{
    int m,n;
    cin >> m >> n;
    cout << lcm(m,n) << endl;
}

排列组合

排列组合也遇到过，解决“有多少种”这类问题经常遇见，只不过，这类问题都很简单，一学就会；概念高中已经学

过了，这里就不多说了；

排列A

#include<iostream>
#include<algorithm>
using namespace std;

int A(int m, int n)
{
    int i,t,ans=1;
    if(m<n)
    {
        m=t;
        t=n;
        n=m;
    }
    for(i=m;i>=m-n+1;i--)
    {
        ans *= i;
    }
    return ans;
}
int main()
{
    int m,n;
    cin >> m >> n;
    if(m>n) cout << A(m,n) << endl;
    else cout << A(n,m) << endl;
}

组合C

#include<iostream>
#include<algorithm>
using namespace std;

int C(int n,int k)
{
    int ans=1;
    int i;
    for(i=n;i>=n-k+1;i--)
    {
        ans*=i;
    }
    for(i=1;i<=k;i++)
    {
        ans/=i;
    }    
    return ans;
}
int main()
{
    int m,n;
    cin >> m >> n;
    if(m>n) cout << C(m,n) << endl;
    else cout << C(n,m) << endl;
}

快速幂

快速幂这个算法可以用来计算那些次方很多数据很大的问题，有些问题会出现TLE，引入快速幂，很好得解决了这

个问题，它的时间复杂度为 O(log₂N)，上代码：

#include<iostream>
#include<algorithm>
using namespace std;

long long quick(long long x,long long y)
{
    long long ans = 1,base = x;
    while( y > 0 )
    {
        if( y % 2 )
        {
            ans *= base;
        }
        base *= base;
        y /= 2;
    }
    return ans;
}
int main()
{
    long long m,n;
    cin >> m >> n;
    cout << quick(m,n) << endl;
}

这个可以计算到2的62次方左右，因为long long的范围有限；

#include<iostream>
#include<algorithm>
#include<cstring>
using namespace std;

int main()
{
    int m=1,n=1,i,j,L1,L2,k;
    char a[1002],b[1002];
    int num1[1002],num2[1002];
    memset(num1,0,sizeof(num1));            //数组清零 
    memset(num2,0,sizeof(num2));            //数组清零
    cin >> a >> b;                              //输入数组 
    L1 = strlen(a);                  //计算数组a的长度 
    L2 = strlen(b);                 //计算数组b的长度 
    for(i=L1-1;i>=0;i--)
    {
        num1[m] = a[i] - '0';   
        m++;                   //字符-->数字 
    }
    for(i=L2-1;i>=0;i--)
    {
        num2[n] = b[i] - '0';    
        n++;                  //字符-->数字 
    }
    for(i=1;i<1002;i++)
    {
        k = i;
        num1[i] += num2[i];        // num1数组当成一个新的数组，重新赋值 
        if(num1[i]>9)

            num1[i] = num1[i]%10;    //判断是否超过10 超过进位

        if(num1[i]!=0)             //判断什么时候不为 0

二分法

这个算法是过双创中心面试之后第一个讲解的算法，其实现在还是懵懵的；

二分查找法，其实是一种有序的查找方法，也称折半查找，如果是无序的则要先进行排序操作；

基本思想是：目标值通过与中间元素比较，可分为三种情况：

1、目标值与中间元素相等，查找结束；

2、目标值比中间元素大，则把后半部分的中间元素与目标值比较；

3、目标值比中间元素小，则把前半部分的中间元素与目标值比较；

这三步一直循环，直到找到答案为止；

适用情况：求最大值中的最小值、求最小值中的最大值；（听着很别扭，其实就是这样）

类似于数学里面的二分，思想差不多；

以一道题来举例：杭电4004

#include<iostream>
#include<algorithm>
using namespace std;

int a[500200];
int L,n,m;
int check(int x) //二分函数
{
    int i,cnt=0,t=0;
    for(i=1;i<=n;i++)
    {
        if(a[i]-a[i-1]>x)
        {
            return 0;
        }
        if(a[i]-t>x)
        {
            cnt++;
            t=a[i-1];
        }
    }
    if(cnt+1<=m)
        return 1;
    else
        return 0;
}
int main()
{
    int mid,i;
    while(cin>>L>>n>>m)
    {
        for(i=1;i<=n;i++)
        {
            cin >> a[i];
        }
        a[0]=0,a[++n]=L;
        sort(a,a+n+1);
        int l=0,r=L,res;
        while(l<=r) //这里的小于等于千万不能忘记等于
        {
            mid=(l+r)/2; //分半
            if(check(mid)) //取左
            {
                res=mid;
                r=mid-1;
            }
            else //取右
            {
                l=mid+1; 
            }
        }
        cout << res <<endl;
    }
    return 0;
}

Dfs & Bfs

相信打过acm比赛的人都知道这两种算法，一种叫做dfs（深度优先搜索），一种叫做bfs（广度优先搜索）；

怎么说呢，折两种算法早就想学了，就是没找到机会，现在正好学习算法，就简单学学吧；

Dfs

深度优先搜索算法（Depth First Search，简称DFS）：一种用于遍历或搜索树或图的算法。 沿着树的深度遍历树

的节点，尽可能深的搜索树的分支，当节点v的所在边都己被探寻过或者在搜寻时结点不满足条件，搜索将回溯到

发现节点v的那条边的起始节点，整个进程反复进行直到所有节点都被访问为止；

模板
int check(参数)
{
    if(满足条件)
    {
        return 1;
    }
    else
    {
        return 0;
    }
}

void dfs(int step)
{
        判断边界
        {
            ...操作
        }
        尝试每一种可能
        {
               满足check条件
               标记
               继续下一步dfs(step+1)
               恢复初始状态（回溯的时候要用到）
        }
}

模板也是根据实际情况来改，下面通过一个实例来了解一下：

全排列问题

问题描述：1~n , n个数字，能组成多少个不同的数字？

例如：n=3，则能组成的数字有：123 , 132 , 213 , 231 , 312 , 321

代码
#include<iostream>
#include<algorithm> 
using namespace std;

int a[10], book[10], n,num=0;

int dfs(int step)
{
    int i;
    if (step == n + 1) //判断边界
    {
        for (i = 1; i <= n; i++)
        {
            cout << a[i];num++;
        }
        cout << endl;
    }
    for (i = 1; i <= n; i++) //遍历
    {
        if (book[i] == 0) //满足check
        {
            a[step] = i;
            book[i] = 1;  //标记
            dfs(step + 1); //继续搜索
            book[i] = 0; //恢复初始状态
        }
    }
    return num;
}

int main()
{
    cin >> n;
    dfs(1); //从第一个开始
    cout << "共有" << num/n << "个";
    return 0;
}

Bfs

广度优先算法（Breadth-First-Search，简称BFS），是一种图形搜索演算法。简单的说，BFS是从根节点开始，沿

着树的宽度遍历树的节点，如果发现目标，则演算终止；
