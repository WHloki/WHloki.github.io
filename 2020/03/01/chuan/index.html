学完了栈与队列，应该都会了吧，今天来聊聊串；

串

定义

串是由零个或多个字符组成的有限序列，又名叫字符串；

串中的字符数目n称为串的长度；

零个字符的串称为空串，可以用两双引号””表示，也可以用希腊字母Ø表示；

字符串，大家都不陌生对吧，学习C语言的时候就讲过了，编程题中常常遇到，这里就不多说了；

串的比较

这里串的比较在C语言也学过了，strcmp()函数，一会会讲到，就不多说；

串的抽象数据类型

1、StrAssign(S, *chars) //生成一个值等于chars的串S

2、StrCopy(S,T) //串T复制得串S

3、StrLength(S) //串S的长度，即S中的元素个数

4、StrCat(S,T) //串T连接在串S后面

5、SubString(Sub,S,pos,len) //用Sub返回串S的第pos个字符起长度为len的子串

6、StrIndex(S,T) //若串S中存在与串T相同的子串，则返回它在串S中第一次出现的位置，否则返回代表错误的值

7、StrInsert(S,pos,T) //在串S的第pos个字符插入串T

8、StrDelete(S,pos,len) //从串S中删除第pos个字符起长度为len的子串

9、StrReplace(S,T,V) //用串V替换串S中出现得到所有与T相等的不重叠子串

10、StrEmpty(S) //若串S为空串，则返回1，否则返回0

11、StrCompare(S,T) //若S>T，则返回值>0,若S=T，则返回值=0,若S<T，则返回值<0

12、StrClear(S) //将串S清空

13、DispStr(S) //显示串S的所有字符

串的存储结构

串的存储结构也是分为顺序存储结构和链式存储结构，重点讲顺序存储结构，即定长顺序串；

结构代码

#define MAXSIZE 100
typedef struct
{
    char ch[MAXSIZE];
    int Len;
}String;

字符串长度

int StrLength(String* S) //求字符串S长度
{
    int i = 0;
    while (S->ch[i] != '\0')
    {
        i++;
    }
    S->Len = i;
    return (S->Len);
}

建立新串

void CreateStr(String* S) //建立一个新串
{
    gets(S->ch);
    S->Len = StrLength(S);
}

求子串函数

int SubString(String* S, String* Sub, int pos, int len) //求子串函数，用Sub返回串S的第pos个字符起长度为len的子串
{
    int j;
    if (pos<1 || pos>S->Len || len<1 || len>S->Len - pos + 1) //范围错误
    {
        Sub->Len = 0;
        printf("ERROR\n");
        return 0;
    }
    else
    {
        for (j = 0; j < len; j++)
        {
            Sub->ch[j] = S->ch[pos + j - 1]; //新串Sub赋值
        }
        Sub->ch[j] = '\0'; //新串尾部加上字符串结束标志
        Sub->Len = len; //规定长度
        return 1;
    }
}

删除子串

int StrDelete(String* S, int i, int l) //删除子串，在S中删除第i开始连续l个字符
{
    int k;
    if (i + l - 1 > S->Len)
    {
        printf("子串超界")
        return 0;
    }
    else
    {
        for (k = i + l - 1; k < S->Len; k++, i++) //从第i位开始删除长度为l个字符
        {
            S->ch[i - 1] = S->ch[k];
        }
        S->Len = S->Len - l; //长度-l
        S->ch[S->Len] = '\0'; //新串尾部加上字符串结束标志

        return 1;
    }
}

插入子串

int StrInsert(String *S,String *T,int i) //串S中第i个字符插入子串T
{
    int k;
    if (i > S->Len + 1)
    {
        printf("插入位置错误")；
        return 0;
    }
    else if(S->Len+T->Len>MAXSIZE)
    {
        printf("两串长度超过存储空间长度")
        return 0；
    }
    else
    {
        for (k = S->Len - 1; k >= i - 1; k--) //将第i位开始的字符各向后移动T串长度
        {
            S->ch[T->Len + k] = S->ch[k];
        }
        for (k = 0; k < T->Len; k++) //将子串T插入到串S的第i位处
        {
            S->ch[i + k - 1] = T->ch[k];
        }
        S->Len = S->Len + T->Len; //修改串S的长度
        S->ch[S->Len] = '\0'; //新串S尾部加上字符串结束标志

        return 1;
    }
}

定位子串

int StrIndex(String* S, String* T) //串的定位函数，若串S中存在与串T相同的子串，则返回它在串S中第一次出现的位置(第几个字符，不是下标)
{
    int i = 0, j = 0, k;
    while (i < S->Len && j < T->Len) //当两串指针没指向该串尾时进行比较
    {
        if (S->ch[i] == T->ch[j])
        {
            i++;
            j++;
        }
        else
        {
            i = i - j + 1;
            j = 0;
        }
    }
    if (j >= T->Len) //串S中有串T
    {
        k = i - T->Len + 1;
    }
    else
    {
        k = -1;
    }
    return k;
}

字符串比较

int StrCompare(String* S1, String* S2)//判断两个串是否相等
{
    int i = 0, flag = 0;
    while (S1->ch[i] != '\0' && S2->ch[i] != '\0') //当两串没到串尾部时
    {
        if (S1->ch[i] != S2->ch[i]) //两串对应位置的字符是否相同
        {
            flag = 1; //标志置为1
            break; //结束循环
        }
        else
        {
            i++; //计数
        }
    }
    if (flag == 0 && S1->Len == S2->Len) //两串长度相同且所有字符都相同，两串相等
    {
        return 0;
    }
    else
    {
        return S1->ch[i] - S2->ch[i]; //返回对应第一个不相同位置字符ASCII码差值
    }
}

字符串连接

int StrCat(String* S, String* T) //连接函数
{
    int i, flag;
    if (S->Len + T->Len <= MAXSIZE) //连接后串长小于MAXSIZE
    {
        for (i = S->Len; i < S->Len + T->Len; i++)
        {
            S->ch[i] = T->ch[i - S->Len]; //赋值连接
        }
        S->ch[i] = '\0';
        S->Len += T->Len; //长度相加
        flag = 1;
    }
    else if (S->Len < MAXSIZE) //连接后串长大于MAXSIZE，但串S长度小于MAXSIZE，连接后串T部分字符序列被舍弃
    {
        for (i = S->Len; i < MAXSIZE; i++)
        {
            S->ch[i] = T->ch[i - S->Len];
        }
        S->Len = MAXSIZE;
        flag = 0;
    }
    else //串S的长度等于MAXSIZE，串T不被连接
    {
        flag = 0;
    }
    return flag;
}

子串替换

void StrReplace(String* S, String* T, String* V) //子串替换，用串V替换串S中出现得到所有与T相等的不重叠子串
{
    int i, m, n, p, q;
    n = S->Len;
    m = T->Len;
    q = V->Len;
    p = 1;
    do {
        i = StrIndex(S, T); //调用定位函数得到子串T在子串S的位置

        if (i != -1) //当主串S有该子串T时
        {
            StrDelete(S, i, m); //删除子串T
            StrInsert(S, V, i); //插入子串V
            p = i + q;
            S->Len = S->Len + q - m; //修改主串S的长度
            n = S->Len;+
        }
    } while ((p <= n - m + 1) && (i != -1));
}
