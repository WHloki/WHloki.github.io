栈与队列—Stack & Queue

学习完线性表，今天开始学习栈与队列；

栈

定义

栈是限定仅在表尾进行插入和删除操作的线性表；

允许插入和删除的一端称为栈顶，另一端称为栈底，不含任何数据元素的栈称为空栈，栈又称为后进先出线性表，

简称LIFO结构；

栈的抽象数据类型——基本操作( 入栈 出栈 )

1、InitStack(*S) //初始化操作，建立一个空栈

2、DestroyStack(*S) //若栈存在，则销毁它

3、ClearStack(*S) //将栈清空

4、StackEmpty(*S) //若栈为空，返回true，否则返回false

5、GetTop(S, *e) //若栈存在且非空，用e返回S的栈顶元素

6、Push(*S, e) //若栈S存在，插入新元素e到栈S中并成为栈顶元素

7、Pop(*S, *e) //删除栈S中栈顶元素，并用e返回其值

8、StackLength(S) //返回栈S的元素个数

栈的顺序存储结构

结构代码

#define MAXSIZE 1000
typedef int DataType;
typedef struct
{
    DataType data[MAXSIZE]; //栈的元素最大个数
    int top; //用于栈顶指针
}SqStack;

初始化栈

void InitStack(SqStack* S) //初始化栈
{
    S->top = -1; //初始化栈为空
}

判断栈是否已满

int StackFull(SqStack* S) //判断栈是否已满
{
    if (S->top == MaxSIZE - 1)
        return 1;
    else
        return 0;
}

判断栈是否为空

int StackEmpty(SqStack* S) //判断栈是否为空
{
    if (S->top == -1)
        return 1
    else
        return 0;
}

进栈

int Push(SqStack* S, DataType e) //进栈操作 插入元素e为新的栈顶元素
{
    if (StackFull(S)) //栈满
    {
        return 0;
    }
    S->top++; //栈顶指针增加1
    S->data[S->top] = e; //新插入元素赋值给栈顶空间
    return 1;
}

出栈

int Pop(SqStack* S, DataType* e) //出栈操作 若栈不为空，则删除栈顶元素，用e返回其值
{
    if (StackEmpty(S)) //栈空
    {
        return 0;
    }
    *e = S->data[S->top]; //将要删除的元素赋值给e
    S->top--; //栈顶指针减少1
    return 1;
}

栈的链式存储结构

结构代码

#define MAXSIZE 100
typedef struct StackNode
{
    DataType data;
    struct StackNode* next;
}LinkStack;

初始化栈

LinkStack* InitStack()
{
    LinkStack* S;
    S = NULL; //初始化栈为空
    return S;
}

判断栈是否为空

int EmptyStack(LinkStack* S)
{
    if (S == NULL)
        return 1;
    else
        return 0;
}

进栈

int Push(LinkStack* S, DataType e)
{
    LinkStack* p = (LinkStack)malloc(sizeof(LinkStack));
    p->data = e; //e放入新节点的数据域
    p->next = S; //新节点插入链表表头之前
    S = p; //新节点作为栈顶
    return 1; 
}

出栈

int Pop(LinkStack* S, DataType* e)
{
    LinkStack* p;
    if (EmptyStack(S))
    {
        return 0;
    }
    *e = S->data; //栈顶元素赋值给e
    p = S; //p节点指向原栈顶S
    S = S->next; //原栈顶S指向其下一个节点
    free(p); //释放原栈顶空间
    return 1;
}

取栈顶元素

int GetTop(LinkStack* S, DataType* e)
{
    if (EmptyStack(S)) //栈为空
    {
        return 0;
    }
    *e = S->data; //栈顶元素赋给变量e
    return 1;
}

遍历栈

void ShowStack(LinkStack* S)
{
    LinkStack* p = S;
    if (p == NULL)
    {
        printf("栈为空\n");
        return;
    }
    while (p != NULL)
    {
        printf("%d", p->data);
        p = p->next;
    }
}

队列

定义

队列是只允许在一端进行插入操作，在另一端进行删除操作的线性表；

队列是一种先进先出的线性表，简称FIFO；允许插入的一端称为队尾，允许删除的一端称为队头；

队列的抽象数据类型——基本操作

1、InitQueue(*Q) //初始化操作，建立一个空队列

2、DestroyQueue(*Q) //若队列Q存在，则销毁

3、ClearQueue(*Q) //将队列Q清空

4、QueueEmpty(Q) //若队列为空，返回true，否则返回false

5、GetHead(Q,*e) //若队列存在且非空，用e返回Q的队头元素

6、EnQueu(*Q,e) //若队列存在，则插入e于Q，并成为队尾元素

7、DeQueue(*Q,*e) // 删除队列Q中队头元素，并用e返回其值

8、QueueLength(Q) //返回队列Q中元素个数

循环队列

队列顺序存储的不足

与栈不同的是，队列的出列是在队头，那就意味着，队列的所有元素都得移动，以保证下标为0的位置不为空，此

时间复杂度为O(n);

为了避免当只有一个元素时，队头和队尾重合使处理变得麻烦，所以引入两个指针，front指针指向队头元素，

rear指针指向队尾元素的下一个位置，这样当front等于rear时，此队列不是还剩一个元素，而是空队列；

假溢出：说白了就是数组末尾元素已满，再向后加就越界，而队列前面还有空位的现象（0,1,2….空着）；

循环队列定义

头尾相接顺序存储结构称为循环队列；

设队列最大长度为QueueSize，那么：

队列满的条件为(rear+1) % QueueSize == front ;

队空条件为front == rear ;

队列长度计算公式为(rear-front+QueueSize) % QueueSize ; (自己推导式子很简单)

循环队列顺序存储结构

#define MAXSIZE 100
typedef int DataType;
typedef struct
{
    DataType data[MAXSIZE];
    int front; //头指针
    int rear; //尾指针，若队列不变，指向队列尾元素的下一个位置
}SqQueue;

队列初始化

int InitQueue(SqQueue *Q) //初始化一个对列
{
    Q->front = 0;
    Q->rear = 0;
    return 1;
}

队列长度

int QueueLength(SqQueue Q) //求队列长度
{
    return (Q.rear - Q.front + MAXSIZE) % MAXSIZE;
}

入队

int EnQueue(SqQueue* Q, DataType e) //若队列未满则e入列
{
    if ((Q->rear + 1) % MAXSIZE == Q->front) //队列已满
    {
        return 0;
    }
    Q->data[Q->rear] = e; //e赋值给队尾
    Q->rear = (Q->rear + 1) % MAXSIZE; //rear指针后移一个位置，若到最后则转到数组头部

    return 1;
}

出队

int DeQueue(SqQueue* Q, DataType* e) //若队列不空则删除队头，并用e返回其值
{
    if (Q->front == Q->near) //队列为空
    {
        return 0;
    }
    *e = Q->data[Q->front]; //将队头元素赋值给e
    Q->front = (Q->front + 1) % MAXSIZE; //front指针向后移一位置，若到最后则转到数组头部

    return 1;
}

队列的链式存储结构及实现

队列的链式存储结构，其实就是线性表的单链表，只不过它只能尾进头出而已，我们把它简称为链队列；

链式队列存储结构

typedef int DataType;
typedef struct qnode
{
    DataType data; //节点的数据域
    struct qnode* next; //节点的指针域
}LinkListQ;
typedef struct
{
    LinkListQ* front, * rear; //队列的队头和队尾指针
}LinkQueue;

队列初始化

LinkQueue* InitQueue() //队列初始化
{
    LinkQueue* Q;
    LinkListQ* p;
    Q = (LinkQueue*)malloc(sizeof(LinkQueue)); //头指针所指节点
    p = (LinkListQ*)malloc(sizeof(LinkListQ)); //头节点
    Q->front = p; //Q指针所指的front指针指向p
    Q->rear = p; //Q指针所指的rear指针指向p
    return Q;
}

入队

int EnQueue(LinkQueue* Q, DataType e) //插入e于队尾
{
    LinkListQ* s;
    s = (LinkListQ)malloc(sizeof(LinkListQ)); //分配内存空间
    s->data = e; //将e存入新节点的数据域
    s->next = NULL;
    Q->rear->next = s; //将新节点插入链队之后
    Q->rear = s; //队尾指针指向队尾元素

    return 1;
}

出队

int DeQueue(LinkQueue* Q, DataType* e) //删除队头，用e返回其值
{
    LinkListQ* p;
    if (Q->front == Q->rear) //队空
    {
        return 0;
    }
    p = Q->front->next; //p指向队头元素
    *e = p->data; //队头元素的值赋给e
    Q->front->next = p->next; //原队头节点后继p->next赋给头节点后继
    if (Q->rear == p)/*p->next==NULL*/ //队头等于队尾，删除后将rear指向头节点
    {
        Q->rear = Q->front; //此时队空
    }
    free(p);

    return 1;
}

小结

对于循环队列和链队列的比较，可以从两方面，从时间上看，它们基本操作都是常数时间，都为O(1)，从空间看，

循环队列必须有一个固定的长度，就有了存储元素个数和空间资源浪费的问题，而链队列不存在这个问题；尽管需

要一个指针域，需要空间，但是可以接受，所以空间上链队列更加灵活；

总的来说，在可以确定队列长度最大值的情况下，建议使用循环队列，无法估计队列长度时则用链队列；
