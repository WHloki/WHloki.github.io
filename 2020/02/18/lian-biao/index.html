链表—Link List

链表是C语言中非常难理解的一个知识点，可以说是比指针还难学的东西。在学习数据结构的时候常常会用到，所

以想来想去有必要整理一波；

链表种类很多种，有单向链表，双向链表，空头链表，非空头链表，循环链表，非循环链表；

这里我只学了单向空头和单向非空头链表，先整理这两种，以后学完再添加；

链表

基本概念

链表实际上是线性表的链式存储结构，与数组不同的是，它是用一组任意的存储单元来存储线性表中的数据，存储

单元不一定是连续的；

且链表的长度不是固定的，链表数据的这一特点使其可以非常的方便地实现节点的插入和删除操作；

链表的每个元素称为一个节点，每个节点都可以存储在内存中的不同的位置，为了表示每个元素与后继元素的逻辑

关系，以便构成“一个节点链着一个节点”的链式存储结构；

除了存储元素本身的信息外，还要存储其直接后继信息，因此，每个节点都包含两个部分，第一部分称为链表的数

据区域，用于存储元素本身的数据信息，第二部分是一个结构体指针，称为链表的指针域，用于存储其直接后继的

节点信息；

next的值实际上就是下一个节点的地址，当前节点为末节点时，next的值设为空指针；

优点（相对于数组）

对线性表进行删除，添加等操作时候，不需要移动大量数据，操作简便，而数组需要动用大量数据，十分困难；同

时链表属于动态分配内存，数据量可以改变，而数组长度必须提前定义好，属于静态分配内存；链表省空间，数组

容易越界，长度过长又会造成系统资源浪费；

总的来说，链表优点非常多；

缺点

单向链表只能顺序访问，不能随机访问，这种储存结构最大缺点就是容易断链；一旦某个节点指针域数据丢失，将

无法找到下一个节点，也就是说，该节点的后面的数据全部丢失；

无空头链表

添加

头添加
void AddHead(int n) //创建链表，在链表中增加一个数据，头添加
{
    //创建一个节点
    struct Node* pTemp = (struct Node*)malloc(sizeof(struct Node));

    //节点数据进行赋值
    pTemp->a = n;
    pTemp->next = NULL;

    //接在链表上
    if (head == NULL)//链表为空，新节点既是头也是尾巴
    {
        head = pTemp;
        tail = pTemp;
    }
    else
    {
        pTemp->next = head; //新节点的下一个指向头
        head = pTemp;//新节点就是头了
    }
}
尾添加
void AddTail(int n) //创建链表，在链表中增加一个数据，尾添加
{
    //创建一个节点
    struct Node* pTemp = (struct Node*)malloc(sizeof(struct Node));

    //节点数据进行赋值
    pTemp->a = n;
    pTemp->next = NULL;

    //链接
    if (head == NULL || tail == NULL)//空链表，既是头，又是尾
    {
        head = pTemp;
    }
    else //不为空，往尾巴上添加
    {
        tail->next = pTemp;
    }
    tail = pTemp;// 尾巴一直指向最后一个节点
}
任意位置添加
void AddList(int index, int n)//在数字index后面添加数字n
{
    if (head == NULL)
    {
        printf("链表为空，没有节点\n");
        return;
    }
    struct Node* fp = SearchNode(index);
    if (fp == NULL)
    {
        printf("没有指定的节点\n");
        return;
    }
    //有此节点，给a创建节点
    struct Node* pTemp = (struct Node*)malloc(sizeof(struct Node));
    pTemp->a = n;
    pTemp->next = NULL;

    if (fp == tail)//链接到链表上
    {
        tail->next = pTemp; //尾巴的下一个，指向新的节点
        tail = pTemp;  //新节点是最后一个，变成尾巴
    }
    else
    {
        pTemp->next = fp->next;//先连
        fp->next = pTemp;      //后断
    }
}

删除

头删除
void DeleteHead() //头删除
{
    if (head == NULL)//检查链表是否为空
    {
        printf("链表为NULL，无需释放!\n");
        return;
    }
    if (head == tail)
    {
        free(head);
        head = NULL;
        tail = NULL;
    }
    else
    {
        struct Node* pTemp = head;//记住旧的头
        head = head->next; //头的下一个节点变成新的头
        free(pTemp); //释放旧的头
    }
}
尾删除
void DeleteTail() //尾删除
{
    if (head == NULL)//检查链表是否为空
    {
        printf("链表为NULL，无需释放!\n");
        return;
    }
    //链表不为空 有一个节点
    if (head == tail)
    {
        free(head);
        head = NULL;
        tail = NULL;
    }
    else
    {
        //找尾巴前一个节点
        struct Node* pTemp = head;
        while (pTemp->next != tail)
        {
            pTemp = pTemp->next;
        }
        //找到尾巴 删除并释放
        free(tail);  //释放尾巴
        tail = pTemp;//移动尾巴
        tail->next = NULL;//尾巴的下一个指针赋值为NULL
    }
}
任意位置删除
void DeleteNode(int n)//删除指定节点 
{
    if (head == NULL)//检查链表是否为空
    {
        printf("链表为NULL，无需释放!\n");
        return;
    }
    //链表有东西，找此节点
    struct Node* pTemp = SearchNode(n);
    if (pTemp == NULL)
    {
        printf("查无此节点\n");
        return;
    }
    //找到了
    if (pTemp == head) //只有一个节点
    {
        DeleteHead();
    }
    else if (pTemp == tail)//有两个节点
    {
        DeleteTail();
    }
    else//有多个节点
    {

        struct Node* fp = head;
        while (fp->next != pTemp)
        {
            fp = fp->next;
        }
        //找到了
        fp->next = pTemp->next;//连接
        free(pTemp); //释放
    }
}

遍历

void SeekList() //遍历链表——>查
{
    struct Node* pTemp = head;
    while (pTemp != NULL)
    {
        printf("%d  ",pTemp->a);
        pTemp = pTemp->next;
    }
    printf("\n");
}

查询

struct Node* SearchNode(int n) //查询指定节点
{
    struct Node* pTemp = head;
    int num = 1;
    while (pTemp != NULL)
    {
        if (n == pTemp->a)
        {
            printf("第%d个数为%d\n", num, pTemp->a);
            //return pTemp;
        }
        pTemp = pTemp->next;
        num++;
    }
    return NULL; //没找到
}

修改

void ChangeNode(int index, int n)//将index改为 n  
{
    struct Node* fp = head->next;
    while (fp != NULL)
    {
        if (fp->a == index)
        {
            fp->a = n;
            return;
        }
        fp = fp->next;
    }
}

清空

void FreeList() //链表清空
{
    struct Node* pTemp = head; //记录头，防止头被修改，丢内存
    while (pTemp != NULL)
    {
        struct Node* fp = pTemp;
        pTemp = pTemp->next;
        free(fp);
    }
    //头尾清空
    head = NULL;
    tail = NULL;
}

空头链表

空头链表和无空头链表其实本质区别不大，空头链表就是头结点为空（有头指针，但是没有数值域）在进行某些操

作的时候就比较简单；

空头链表和无空头链表相似度很高，部分操作相同，就不一一列举了；

空头链表初始化

void NewList()//空头链表初始化
{
    //链表空头
    head = (struct Node*)malloc(sizeof(struct Node));
    head->next = NULL;
    tail = head;
}

创建节点

struct Node* CreateNode(int n)//创建节点
{
    struct Node* pTemp = (struct Node*)malloc(sizeof(struct Node));//创建一个节点
    if (pTemp == NULL)//创建失败
    {
        return NULL;
    }

    pTemp->a = n; //节点成员赋值
    pTemp->next = NULL;
    return pTemp;//返回节点
}

添加

头添加
void AddHead(int n) // 头添加
{
    struct Node* pTemp = CreateNode(n);//创建一个节点

    if (pTemp == NULL)//创建失败
    {
        printf("内存不足\n");// 一般不会申请失败，除非内存不足
        return;
    }
    //连接
    pTemp->next = head->next;
    head->next = pTemp;
}
尾添加
void AddTail(int n) //尾添加
{
    struct Node* pTemp = CreateNode(n);//创建一个节点

    if (pTemp == NULL)//创建失败
    {
        printf("内存不足\n");// 一般不会申请失败，除非内存不足
        return;
    }
    //连接
    tail->next = pTemp;
    tail = pTemp;
}
任意位置添加
void AddRand(int index,int n)//在数字index后面添加数字a
{
    if (head->next == NULL)//判断链表有没有东西
    {
        printf("链表无节点\n");
        return;
    }
    //找index节点
    struct Node* pTemp = SearchNode(index);
    if (pTemp == NULL)
    {
        printf("查无此节点\n");
        return;
    }
    //找到节点,连接到指定位置
    //创建节点
    struct Node* pNode = CreateNode(n);
    //连接
    if (pTemp == tail)
    {
        AddTail(n);
    }
    else
    {
        pNode->next = pTemp->next;
        pTemp->next = pNode;
    }
}

删除

头删除
void DeleteHead()//删除头
{
    if (head->next == NULL)//判断链表有没有东西
    {
        printf("链表无头"); 
        return;
    }
    //有节点 ——> 删除  
    struct Node* pTemp = head->next; //记录头
    head->next = head->next->next;  //变头

    free(pTemp);//释放
}
尾删除
void DeleteTail()//删除尾
{ 
    if (head->next == NULL)//判断链表有没有东西
    {
        printf("链表无尾");
        return;
    }
    //有一个节点 ——> 删除
    if (head->next == tail)
    {
        free(tail);
        head->next = NULL;
        tail = head;
    }
    else
    {
        struct Node* pTemp = head->next;
        while (pTemp != NULL)
        {
            if(pTemp->next == tail)
            { 
                break;
            }
            pTemp = pTemp->next;
        }
        //pTemp就是尾巴的前一个

        free(tail); //释放尾巴
        tail = pTemp; //更新尾巴
        tail->next = NULL; //尾巴下一个赋值为NULL
    }
}
